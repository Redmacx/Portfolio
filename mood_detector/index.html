🐍 Python Script & Download

Run this script locally on your computer with a webcam using OpenCV Haar Cascades.

import cv2 # Load OpenCV Haar Cascades face_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_frontalface_default.xml') smile_cascade = cv2.CascadeClassifier(cv2.data.haarcascades + 'haarcascade_smile.xml') cap = cv2.VideoCapture(0) while True: ret, frame = cap.read() gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY) faces = face_cascade.detectMultiScale(gray, 1.2, 5) for (x, y, w, h) in faces: roi_gray = gray[y:y+h, x:x+w] smiles = smile_cascade.detectMultiScale(roi_gray, 1.7, 20) mood = "HAPPY 😄" if len(smiles) > 0 else "NEUTRAL / FOCUSED 😐" color = (0, 255, 0) if len(smiles) > 0 else (255, 165, 0) cv2.rectangle(frame, (x, y), (x+w, y+h), color, 2) cv2.putText(frame, mood, (x, y-10), cv2.FONT_HERSHEY_SIMPLEX, 0.7, color, 2) cv2.imshow("Mood Detector", frame) if cv2.waitKey(1) & 0xFF == ord('q'): break cap.release() cv2.destroyAllWindows()
